120  */
121 void EK_TM4C1294XL_initDMA(void)
122 {
123  Error_Block eb;
124  Hwi_Params hwiParams;
125 
126  if (!dmaInitialized) {
127  Error_init(&eb);
128  Hwi_Params_init(&hwiParams);
129  Hwi_construct(&(dmaHwiStruct), INT_UDMAERR, dmaErrorHwi,
130  &hwiParams, &eb);
131  if (Error_check(&eb)) {
132  System_abort("Couldn't construct DMA error hwi");
133  }
134 
135  SysCtlPeripheralEnable(SYSCTL_PERIPH_UDMA);
136  uDMAEnable();
137  uDMAControlBaseSet(dmaControlTable);
138 
139  dmaInitialized = true;
140  }

231 /*
232  * ======== EK_TM4C1294XL_initEMAC ========
233  */
234 void EK_TM4C1294XL_initEMAC(void)
235 {
236  uint32_t ulUser0, ulUser1;
237 
238  /* Get the MAC address */
239  FlashUserGet(&ulUser0, &ulUser1);
240  if ((ulUser0 != 0xffffffff) && (ulUser1 != 0xffffffff)) {
241  System_printf("Using MAC address in flash\n");
242  /*
243  * Convert the 24/24 split MAC address from NV ram into a 32/16 split MAC
244  * address needed to program the hardware registers, then program the MAC
245  * address into the Ethernet Controller registers.
246  */
247  macAddress[0] = ((ulUser0 >> 0) & 0xff);
248  macAddress[1] = ((ulUser0 >> 8) & 0xff);
249  macAddress[2] = ((ulUser0 >> 16) & 0xff);
250  macAddress[3] = ((ulUser1 >> 0) & 0xff);
251  macAddress[4] = ((ulUser1 >> 8) & 0xff);
252  macAddress[5] = ((ulUser1 >> 16) & 0xff);
253  }
254  else if (macAddress[0] == 0xff && macAddress[1] == 0xff &&
255  macAddress[2] == 0xff && macAddress[3] == 0xff &&
256  macAddress[4] == 0xff && macAddress[5] == 0xff) {
257  System_abort("Change the macAddress variable to match your boards MAC sticker");
258  }
259 
260  GPIOPinConfigure(GPIO_PF0_EN0LED0); /* EK_TM4C1294XL_USR_D3 */
261  GPIOPinConfigure(GPIO_PF4_EN0LED1); /* EK_TM4C1294XL_USR_D4 */
262  GPIOPinTypeEthernetLED(GPIO_PORTF_BASE, GPIO_PIN_0 | GPIO_PIN_4);
263 
264 
265  /* Once EMAC_init is called, EMAC_config cannot be changed */
266  EMAC_init();
267 }

377 {
378  /* I2C7 Init */
379  /*
380  * NOTE: TI-RTOS examples configure pins PD0 & PD1 for SSI2 or I2C7. Thus,
381  * a conflict occurs when the I2C & SPI drivers are used simultaneously in
382  * an application. Modify the pin mux settings in this file and resolve the
383  * conflict before running your the application.
384  */
385  /* Enable the peripheral */
386  SysCtlPeripheralEnable(SYSCTL_PERIPH_I2C7);
387 
388  /* Configure the appropriate pins to be I2C instead of GPIO. */
389  GPIOPinConfigure(GPIO_PD0_I2C7SCL);
390  GPIOPinConfigure(GPIO_PD1_I2C7SDA);
391  GPIOPinTypeI2CSCL(GPIO_PORTD_BASE, GPIO_PIN_0);
392  GPIOPinTypeI2C(GPIO_PORTD_BASE, GPIO_PIN_1);
393 
394  /* I2C8 Init */
395  /* Enable the peripheral */
396  SysCtlPeripheralEnable(SYSCTL_PERIPH_I2C8);
397 
398  /* Configure the appropriate pins to be I2C instead of GPIO. */
399  GPIOPinConfigure(GPIO_PA2_I2C8SCL);
400  GPIOPinConfigure(GPIO_PA3_I2C8SDA);
401  GPIOPinTypeI2CSCL(GPIO_PORTA_BASE, GPIO_PIN_2);
402  GPIOPinTypeI2C(GPIO_PORTA_BASE, GPIO_PIN_3);
403 
404  I2C_init();
405 }

752 /*
753  * ======== EK_TM4C1294XL_initUSB ========
754  * This function just turns on the USB
755  */
756 void EK_TM4C1294XL_initUSB(EK_TM4C1294XL_USBMode usbMode)
757 {
758  Error_Block eb;
759  Hwi_Params hwiParams;
760 
761  /* Enable the USB peripheral and PLL */
762  SysCtlPeripheralEnable(SYSCTL_PERIPH_USB0);
763  SysCtlUSBPLLEnable();
764 
765  /* Setup pins for USB operation */
766  GPIOPinTypeUSBAnalog(GPIO_PORTB_BASE, GPIO_PIN_0 | GPIO_PIN_1);
767  GPIOPinTypeUSBAnalog(GPIO_PORTL_BASE, GPIO_PIN_6 | GPIO_PIN_7);
768 
769  /* Additional configurations for Host mode */
770  if (usbMode == EK_TM4C1294XL_USBHOST) {
771  /* Configure the pins needed */
772  HWREG(GPIO_PORTD_BASE + GPIO_O_LOCK) = GPIO_LOCK_KEY;
773  HWREG(GPIO_PORTD_BASE + GPIO_O_CR) = 0xff;
774  GPIOPinConfigure(GPIO_PD6_USB0EPEN);
775  GPIOPinTypeUSBDigital(GPIO_PORTD_BASE, GPIO_PIN_6 | GPIO_PIN_7);
776 
777  /*
778  * USB bus fault is routed to pin PQ4. We create a Hwi to allow us
779  * to detect power faults and recover gracefully or terminate the
780  * program. PQ4 is active low; set the pin as input with a weak
781  * pull-up.
782  */
783  GPIOPadConfigSet(GPIO_PORTQ_BASE, GPIO_PIN_4,
784  GPIO_STRENGTH_2MA, GPIO_PIN_TYPE_STD_WPU);
785  GPIOIntTypeSet(GPIO_PORTQ_BASE, GPIO_PIN_4, GPIO_FALLING_EDGE);
786  GPIOIntClear(GPIO_PORTQ_BASE, GPIO_PIN_4);
787 
788  /* Create a Hwi for PQ4 pin. */
789  Error_init(&eb);
790  Hwi_Params_init(&hwiParams);
791  Hwi_construct(&(usbBusFaultHwiStruct), INT_GPIOQ4,
792  EK_TM4C1294XL_usbBusFaultHwi, &hwiParams, &eb);
793  if (Error_check(&eb)) {
794  System_abort("Couldn't construct USB bus fault hwi");
795  }
796  }
797 }
